Pow(x, n)

Implement pow(x, n).


class Solution {
public:
    double pow(double x, int n) {
        if(n < 0) {
            if(n == -2147483648) {
                return 1 / (x * pow(x, -(n + 1)));
            }
            double result = 1 / pow(x, -n);
            return result;
        } 
        else {
            if (n == 1) {
                return x;
            }
            else if(n == 2) {
                return x * x;
            }
            else if (n == 0) {
                return 1;
            }
            else if(n % 2 != 0) {
                return x * pow(x, n - 1);
            }
            else {// n is even
                double temp = pow(x, n / 2);
                return temp * temp;
            }
        }
        
    }
};
